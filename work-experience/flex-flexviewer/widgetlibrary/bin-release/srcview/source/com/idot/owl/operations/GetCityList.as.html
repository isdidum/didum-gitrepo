package com.idot.owl.operations
{
    public class GetCityList
    {
        import com.idot.owl.Owl;
        
        import mx.collections.ArrayCollection;
        import mx.rpc.events.ResultEvent;
        protected var ws:Owl = new Owl();
        [Bindable]public var names:ArrayCollection = new ArrayCollection();
        public var numbers:ArrayCollection = new ArrayCollection();
        public var year:String = "";
        
        public function GetCityList()
        {
            names.addItem("fetching...");
            numbers.addItem("fetching...");
            ws.GetCityList(year);
            ws.addEventListener(ResultEvent.RESULT,onServiceResult);
        }

        private function onServiceResult(event:ResultEvent):void
        {
            names = event.result.CityList.City.CityName;
            numbers = event.result.CityList.City.Number;
        }
    }
}
